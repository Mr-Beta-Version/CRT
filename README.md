# CRT SH SUBDOMAINS FINDER

# Install

```
cd $HOME

git clone --depth=1 https://github.com/Mr-Beta-Version/CRT

cd CRT

sudo cp crtsh.py $PREFIX/bin/crtsh

sudo chmod +x $PREFIX/bin/crtsh

```

# Uses

```
crtsh -h
crtsh -d domain.com
crtsh -f domains.txt
crtsh -f domains.txt -s subdomains.txt

```
